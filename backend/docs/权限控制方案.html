1. 核心原则

原则 说明
作业区归属 作业区只属于甲方公司（work_areas.company_id → 甲方公司）
作业区管理权 仅系统管理员 + 甲方管理员可创建/删除/修改/查看作业区
用户管理权 各公司管理员（甲方/乙方/监理方）管理本公司用户（查看/创建/修改/删除）
个人中心 所有用户都有个人中心权限（查看/编辑个人资料、修改密码）

2. 权限模型架构

┌─────────────────────────────────────────────────────────────────┐ │ 权限控制架构 │ ├─────────────────────────────────────────────────────────────────┤ │ │ │ 用户 (User) │ │ │ │ │ ├──→ 角色 (Role) ──→ 角色权限 (Role_Permission) ──→ API 权限 │ │ │ │ │ │ │ └──→ 角色菜单权限 (Role_Menu_Permission) │ │ │ │ │ └──→ 用户自定义权限 (User_Permission) [覆盖角色默认] │ │ │ │ 数据权限 (ABAC) │ │ │ │ │ └──→ 用户作业区 (User_Work_Area) ──→ 数据过滤规则 │ │ │ └─────────────────────────────────────────────────────────────────┘

设计原则

  1. API 即权限：每个 API 端点对应一个权限代码
  2. 前端元素绑定 API：按钮/操作对应到具体 API 权限
  3. 角色默认授权：角色预置一组 API 权限
  4. 用户可覆盖：可为单个用户增删权限（优先级高于角色）
  5. 菜单权限独立：菜单可见性由菜单权限控制
  6. 数据权限补充：甲方作业区用户只能访问绑定的作业区数据

3.2 新增表 1：API 权限映射表

CREATE TABLE api_permissions (
    id              BIGSERIAL PRIMARY KEY,
    api_path        VARCHAR(255) NOT NULL,      -- API 路径，如 /user/list
    api_method      VARCHAR(20)  NOT NULL,      -- HTTP 方法，如 GET/POST
    permission_id   INTEGER      NOT NULL,      -- 关联 permissions 表
    description     VARCHAR(500),               -- 描述
    is_public       BOOLEAN      DEFAULT false, -- 是否公开（无需权限）
    created_at      TIMESTAMP    DEFAULT CURRENT_TIMESTAMP,
    updated_at      TIMESTAMP    DEFAULT CURRENT_TIMESTAMP,
    UNIQUE (api_path, api_method)
);

3.3 新增表 2：UI 元素权限映射表

CREATE TABLE ui_element_permissions (
    id              BIGSERIAL PRIMARY KEY,
    page_code       VARCHAR(50)  NOT NULL,      -- 页面编码
    element_type    VARCHAR(20)  NOT NULL,      -- 元素类型：button/select/menu/item
    element_id      VARCHAR(100) NOT NULL,      -- 元素 ID/标识
    element_name    VARCHAR(200) NOT NULL,      -- 元素名称
    permission_id   INTEGER      NOT NULL,      -- 关联 permissions 表
    action_type     VARCHAR(20)  DEFAULT 'write',
    visible_default BOOLEAN      DEFAULT true,
    enabled_default BOOLEAN      DEFAULT true,
    created_at      TIMESTAMP    DEFAULT CURRENT_TIMESTAMP,
    UNIQUE (page_code, element_type, element_id)
);

7.3 数据权限实现示例（Java）

public List<WorkArea> getAccessibleWorkAreas(User user) {
    String sql = "SELECT wa.* FROM work_areas wa";
    Map<String, Object> params = new HashMap<>();
    
    // 系统管理员：不过滤
    if (user.hasRole("ROLE_SYSTEM_ADMIN")) {
        return jdbcTemplate.query(sql, params, workAreaRowMapper);
    }
    
    // 甲方管理员：按公司过滤（本公司的作业区）
    if (user.getRoleCode().equals("ROLE_JIAFANG_ADMIN")) {
        sql += " WHERE wa.company_id = :companyId";
        params.put("companyId", user.getCompanyId());
        return jdbcTemplate.query(sql, params, workAreaRowMapper);
    }
    
    // 其他角色：无权限访问作业区
    return new ArrayList<>();
}

8.3 Controller 示例

@RestController
@RequestMapping("/user")
public class UserController {
    
    @GetMapping("/list")
    @ApiPermission("user:list")
    @DataScope("USER")
    public Result<Page<UserDTO>> queryUsers(
            @ModelAttribute UserQueryRequest query,
            @RequestAttribute("userId") Long operatorId) {
        Page<UserDTO> page = userService.queryUsers(query, operatorId);
        return Result.success(page);
    }
    
    @PostMapping
    @ApiPermission("user:create")
    public Result<UserDTO> createUser(
            @RequestAttribute("userId") Long operatorId,
            @Valid @RequestBody CreateUserRequest request) {
        UserDTO user = userService.createUser(request, operatorId);
        return Result.success(user);
    }
}

8.4 权限拦截器

@Component
public class PermissionInterceptor implements HandlerInterceptor {
    
    @Autowired
    private PermissionService permissionService;
    
    @Override
    public boolean preHandle(HttpServletRequest request, 
                            HttpServletResponse response, 
                            Object handler) {
        User user = SecurityContextHolder.getCurrentUser();
        String apiPath = request.getRequestURI();
        String method = request.getMethod();
        
        boolean hasPermission = permissionService.hasApiPermission(
            user, apiPath, method);
        
        if (!hasPermission) {
            response.setStatus(403);
            response.getWriter().write("无权限访问");
            return false;
        }
        
        return true;
    }
}

9. 前端实现方案

9.1 权限指令（Vue）

// directives/permission.js
export default {
  mounted(el, binding) {
    const { value } = binding;
    const userStore = useUserStore();
    const permissions = userStore.permissions || [];
    
    if (value && !permissions.includes(value)) {
      el.parentNode && el.parentNode.removeChild(el);
    }
  }
};

// 使用
// <el-button v-permission="'user:create'">创建用户</el-button>

9.2 路由守卫

router.beforeEach((to, from, next) => {
  const userStore = useUserStore();
  const menus = userStore.menus || [];
  const menuCode = to.meta.menuCode;
  
  // 个人中心始终可访问
  if (menuCode === 'profile') {
    next();
    return;
  }
  
  // 检查菜单权限
  const hasPermission = menus.some(m => m.menuCode === menuCode);
  
  if (!hasPermission) {
    next({ name: 'Home' });
    return;
  }
  
  next();
});
